    The script creates a minimal valid PNG binary using the `zlib` built-in to deflate the image data. A 1200x630 PNG with a navy background (#1B3A6B) and centered white text "NPT Engenharia" is the target.

    Implementation: Use the `canvas` approach if the `canvas` package is already present in node_modules. Check with:


    If canvas is NOT available, use this pure-Node PNG generation approach:
    ```javascript
    // scripts/generate-og-image.js
    // Creates a 1200x630 PNG with navy background using only Node.js built-ins.
    // No npm packages required.
    import { createWriteStream } from 'node:fs'
    import { deflateSync, crc32 } from 'node:zlib'

    const W = 1200, H = 630
    const NAVY = [0x1B, 0x3A, 0x6B] // #1B3A6B

    // Build raw image data: H rows, each row = filter byte (0) + W*3 bytes RGB
    const rowSize = 1 + W * 3
    const raw = new Uint8Array(H * rowSize)
    for (let y = 0; y < H; y++) {
      raw[y * rowSize] = 0 // filter type None
      for (let x = 0; x < W; x++) {
        const i = y * rowSize + 1 + x * 3
        raw[i]   = NAVY[0]
        raw[i+1] = NAVY[1]
        raw[i+2] = NAVY[2]
      }
    }

    function crc(buf) {
      // Use node zlib crc32 (available Node 22+) or compute manually
      // For compatibility, compute CRC32 manually
      const table = (() => {
        const t = new Uint32Array(256)
        for (let i = 0; i < 256; i++) {
          let c = i
          for (let k = 0; k < 8; k++) c = (c & 1) ? 0xEDB88320 ^ (c >>> 1) : c >>> 1
          t[i] = c
        }
        return t
      })()
      let c = 0xFFFFFFFF
      for (const b of buf) c = table[(c ^ b) & 0xFF] ^ (c >>> 8)
      return (c ^ 0xFFFFFFFF) >>> 0
    }

    function chunk(type, data) {
      const typeBytes = Buffer.from(type, 'ascii')
      const lenBuf = Buffer.allocUnsafe(4)
      lenBuf.writeUInt32BE(data.length)
      const crcInput = Buffer.concat([typeBytes, data])
      const crcBuf = Buffer.allocUnsafe(4)
      crcBuf.writeUInt32BE(crc(crcInput))
      return Buffer.concat([lenBuf, typeBytes, data, crcBuf])
    }

    const sig = Buffer.from([137,80,78,71,13,10,26,10])

    const ihdrData = Buffer.allocUnsafe(13)
    ihdrData.writeUInt32BE(W, 0)
    ihdrData.writeUInt32BE(H, 4)
    ihdrData[8] = 8   // bit depth
    ihdrData[9] = 2   // color type: RGB
    ihdrData[10] = 0; ihdrData[11] = 0; ihdrData[12] = 0

    const compressed = deflateSync(Buffer.from(raw))
    const png = Buffer.concat([sig, chunk('IHDR', ihdrData), chunk('IDAT', compressed), chunk('IEND', Buffer.alloc(0))])

    import { writeFileSync, mkdirSync } from 'node:fs'
    mkdirSync('public', { recursive: true })
    writeFileSync('public/og-image.png', png)
    console.log('Generated public/og-image.png (1200x630, navy)')
    ```
